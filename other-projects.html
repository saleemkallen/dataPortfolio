-- Advanced vane profile calculations
function calculate_vane_profile(radius, angle, thickness)
    local profile_points = {}
    
    -- Calculate leading edge
    local leading_edge_angle = angle - thickness_angle/2
    local trailing_edge_angle = angle + thickness_angle/2
    
    -- Generate profile points
    for i = 1, num_steps do
        local t = (i - 1) / (num_steps - 1)
        local current_angle = leading_edge_angle + t * (trailing_edge_angle - leading_edge_angle)
        
        local x = radius * math.cos(current_angle)
        local y = radius * math.sin(current_angle)
        
        table.insert(profile_points, v(x, y, 0))
    end
    
    return profile_points
end

function create_multi_vane_impeller(num_vanes)
    local impeller_geometry = {}

        local vane_angle = (vane - 1) * (360 / num_vanes)

end

-- Function to calculate gear geometry
function calculate_gear_geometry(teeth_count, module, pressure_angle)
    local pitch_diameter = teeth_count * module
    local base_diameter = pitch_diameter * math.cos(math.rad(pressure_angle))
    local addendum = module
    local dedendum = 1.25 * module
    local outer_diameter = pitch_diameter + 2 * addendum
    local root_diameter = pitch_diameter - 2 * dedendum
    
    return {
        pitch_diameter = pitch_diameter,
        base_diameter = base_diameter,
        outer_diameter = outer_diameter,
        root_diameter = root_diameter,
        addendum = addendum,
        dedendum = dedendum
    }
end

-- Create gear tooth profile
function create_gear_tooth_profile(gear_params)
    local tooth_points = {}
    local num_points = 50
    
    for i = 1, num_points do
        local t = (i - 1) / (num_points - 1)
        local angle = t * (2 * math.pi / gear_teeth_count)
        
        -- Involute curve calculation
        local r = gear_params.base_diameter / 2
        local x = r * (math.cos(angle) + angle * math.sin(angle))
        local y = r * (math.sin(angle) - angle * math.cos(angle))
        
        table.insert(tooth_points, v(x, y, 0))
    end
    
    return tooth_points
end

-- Create complete gear
function create_gear(teeth_count, module, pressure_angle, width)
    local gear_params = calculate_gear_geometry(teeth_count, module, pressure_angle)
    local gear_geometry = {}
    
    -- Create all teeth
    for tooth = 1, teeth_count do
        local tooth_angle = (tooth - 1) * (2 * math.pi / teeth_count)
        local tooth_profile = create_gear_tooth_profile(gear_params)
        
        -- Rotate tooth to correct position
        for _, point in ipairs(tooth_profile) do
            local x = point.x * math.cos(tooth_angle) - point.y * math.sin(tooth_angle)
            local y = point.x * math.sin(tooth_angle) + point.y * math.cos(tooth_angle)
            table.insert(gear_geometry, v(x, y, 0))
        end
    end
    
    return gear_geometry
end

-- Main gear creation
main_gear = create_gear(gear_teeth_count, gear_module, gear_pressure_angle, gear_width)

function [neutAxis,z]=NeuturalAxis(N,b,E,Layer)
% Calculate neutral axis position for multi-layer beam
% N: number of layers
% b: width of each layer
% E: Young's modulus of each layer
% Layer: thickness of each layer

% Initialize
z=zeros(1,N);
neutAxis=0;

% Calculate layer positions
z(1)=0;
for i=2:N
    z(i)=z(i-1)+Layer(i-1);
end

% Calculate neutral axis position
% Using parallel axis theorem for composite beam
A_total=0;
y_total=0;

for i=1:N-1
    A_i=b(i)*Layer(i);
    y_i=z(i)+Layer(i)/2;
    A_total=A_total+A_i;
    y_total=y_total+A_i*y_i;
end

neutAxis=y_total/A_total;

% Calculate distance from neutral axis for each layer
for i=1:N-1
    z(i)=z(i)+Layer(i)/2-neutAxis;
end

end
function [RohBeam,tTotal,VolFractions]=RohMixture(b,Roh,Layer)
% Calculate effective density of multi-layer beam
% b: width of each layer
% Roh: density of each layer
% Layer: thickness of each layer

N=length(b);
tTotal=sum(Layer);
VolFractions=zeros(1,N);

% Calculate volume fractions
for i=1:N
    VolFractions(i)=b(i)*Layer(i)/(sum(b.*Layer));
end

% Calculate effective density using rule of mixtures
RohBeam=0;
for i=1:N
    RohBeam=RohBeam+VolFractions(i)*Roh(i);
end

end
function [E_eff]=EStiffness(b,E,Layer)
% Calculate effective Young's modulus for multi-layer beam
% b: width of each layer
% E: Young's modulus of each layer
% Layer: thickness of each layer

N=length(b);
A_total=sum(b.*Layer);

% Calculate effective modulus using parallel spring model
E_eff=0;
for i=1:N
    A_i=b(i)*Layer(i);
    E_eff=E_eff+E(i)*A_i/A_total;
end
