            "multi_match": {
                "query": query,
                "fields": ["body_content"],

            }
        },
        "size": 5,
        "_source": ["body_content", "url", "title"]
    }
    
    try:
        response = es_client.search(index="on_prem_test", body=es_query)

def generate_response(query, context_docs):
    """
    Generates response using Azure OpenAI with RAG context.
    """
    context = "\n\n".join([doc['_source']['body_content'] for doc in context_docs])
    
    prompt = f"""Based on the following context, please answer the user's question accurately and concisely.

Context:
{context}

User Question: {query}

Answer:"""

    try:
        response = client.chat.completions.create(
            model=AZURE_DEPLOYMENT_NAME,
            messages=[
                {"role": "system", "content": "You are a helpful assistant that answers questions based on the provided context. Be accurate and concise."},
                {"role": "user", "content": prompt}
            ],
            max_tokens=500,
            temperature=0.7
        )
        
        return response.choices[0].message.content
    except Exception as e:
        print(f"OpenAI error: {e}")
        return "I'm sorry, I couldn't generate a response at this time."

import numpy as np
from sentence_transformers import SentenceTransformer

class TextVectorizer:
    def __init__(self, model_name='all-MiniLM-L6-v2'):
        self.model = SentenceTransformer(model_name)
    
    def vectorize_text(self, text):
        """Convert text to vector representation"""
        return self.model.encode(text)
    
    def find_similar(self, query_vector, document_vectors, top_k=5):
        """Find most similar documents to query"""
        similarities = np.dot(document_vectors, query_vector)
        top_indices = np.argsort(similarities)[-top_k:][::-1]
        return top_indices, similarities[top_indices]

import requests
from bs4 import BeautifulSoup
import time
from urllib.parse import urljoin, urlparse

class WebScraper:
    def __init__(self, delay=1):
        self.delay = delay
        self.session = requests.Session()
    
    def scrape_url(self, url):
        """Scrape content from a single URL"""
        try:
            response = self.session.get(url, timeout=10)
            response.raise_for_status()
            
            soup = BeautifulSoup(response.content, 'html.parser')
            
            # Extract text content
            text_content = soup.get_text(separator=' ', strip=True)
            
            return {
                'url': url,
                'title': soup.title.string if soup.title else '',
                'content': text_content,
                'status': 'success'
            }
            
        except Exception as e:
            return {
                'url': url,
                'error': str(e),
                'status': 'error'
            }
    
    def scrape_multiple(self, urls):
        """Scrape multiple URLs with delay"""
        results = []
        for url in urls:
            result = self.scrape_url(url)
            results.append(result)
            time.sleep(self.delay)
        return results

# Function to calculate NDVI
calculate_ndvi <- function(raster_stack) {
    # NDVI = (NIR - Red) / (NIR + Red)
    # For Landsat 8: NIR = Band 5, Red = Band 4
    ndvi <- (raster_stack[[5]] - raster_stack[[4]]) / (raster_stack[[5]] + raster_stack[[4]])
    return(ndvi)
}

# Function for advanced NDVI analysis
analyze_vegetation_health <- function(ndvi_raster, year) {
    # Calculate vegetation health metrics
    mean_ndvi <- cellStats(ndvi_raster, mean, na.rm = TRUE)
    sd_ndvi <- cellStats(ndvi_raster, sd, na.rm = TRUE)
    
    # Classify vegetation health
    # NDVI > 0.6: Healthy vegetation
    # NDVI 0.3-0.6: Moderate vegetation
    # NDVI < 0.3: Poor vegetation
    healthy_veg <- ndvi_raster > 0.6
    moderate_veg <- ndvi_raster >= 0.3 & ndvi_raster <= 0.6
    poor_veg <- ndvi_raster < 0.3
    
    # Calculate percentages
    total_pixels <- ncell(ndvi_raster[!is.na(ndvi_raster)])
    healthy_pct <- (cellStats(healthy_veg, sum, na.rm = TRUE) / total_pixels) * 100
    moderate_pct <- (cellStats(moderate_veg, sum, na.rm = TRUE) / total_pixels) * 100
    poor_pct <- (cellStats(poor_veg, sum, na.rm = TRUE) / total_pixels) * 100
    
    return(data.frame(
        Year = year,
        Mean_NDVI = mean_ndvi,
        SD_NDVI = sd_ndvi,
        Healthy_Vegetation_Pct = healthy_pct,
        Moderate_Vegetation_Pct = moderate_pct,
        Poor_Vegetation_Pct = poor_pct
    ))
}

# Function for land use classification
perform_land_classification <- function(raster_stack, training_data, year) {
    # Prepare training data
    training_data$class_id <- as.factor(training_data$class_id)
    
    # Extract spectral values for training points
    train_values <- extract(raster_stack, training_data)
    train_df <- data.frame(class = training_data$class_id, train_values)
    
    # Remove NA values
    train_df <- na.omit(train_df)
    
    # Split data for validation
    set.seed(123)
    train_index <- createDataPartition(train_df$class, p = 0.7, list = FALSE)
    train_data <- train_df[train_index, ]
    test_data <- train_df[-train_index, ]
    
    # Train Random Forest model
    rf_model <- randomForest(class ~ ., data = train_data, 
                           ntree = 500, importance = TRUE)
    
    # Validate model
    predictions <- predict(rf_model, test_data)
    confusion_matrix <- confusionMatrix(predictions, test_data$class)
    
    # Print model performance
    print(paste("Year:", year))
    print(paste("Overall Accuracy:", round(confusion_matrix$overall[1], 3)))
    print(paste("Kappa:", round(confusion_matrix$overall[2], 3)))
    
    # Apply classification to entire study area
    land_classification <- predict(raster_stack, rf_model, type = "response")
    
    return(list(
        model = rf_model,
        classification = land_classification,
        accuracy = confusion_matrix$overall[1],
        kappa = confusion_matrix$overall[2]
    ))
}

# Calculate land use change
calculate_land_use_change <- function(class_1, class_2, year1, year2) {
    # Create change matrix
    change_matrix <- table(values(class_1), values(class_2))
    
    # Calculate transition probabilities
    transition_probs <- prop.table(change_matrix, 1)
    
    # Calculate net change for each class
    class_1_freq <- freq(class_1)
    class_2_freq <- freq(class_2)
    
    net_change <- data.frame(
        Class = c("Urban", "Forest", "Agriculture", "Water", "Grassland", "Bare Soil"),
        Year1_Area = class_1_freq$count,
        Year2_Area = class_2_freq$count,
        Net_Change = class_2_freq$count - class_1_freq$count,
        Percent_Change = ((class_2_freq$count - class_1_freq$count) / class_1_freq$count) * 100
    )
    
    return(list(
        change_matrix = change_matrix,
        transition_probs = transition_probs,
        net_change = net_change
    ))
}

analyze_drought_impact <- function(ts_data, species_name) {
    # Calculate growth rates
    ts_data$growth_rate <- c(NA, diff(ts_data$ring_width))
    
    # Analyze 1976 drought impact
    drought_1976_data <- ts_data[ts_data$year %in% 1974:1978, ]
    pre_drought_1976 <- mean(drought_1976_data$growth_rate[1:2], na.rm = TRUE)
    post_drought_1976 <- mean(drought_1976_data$growth_rate[4:5], na.rm = TRUE)
    recovery_1976 <- (post_drought_1976 - pre_drought_1976) / pre_drought_1976 * 100
    
    # Analyze 2003 drought impact
    drought_2003_data <- ts_data[ts_data$year %in% 2001:2005, ]
    pre_drought_2003 <- mean(drought_2003_data$growth_rate[1:2], na.rm = TRUE)
    post_drought_2003 <- mean(drought_2003_data$growth_rate[4:5], na.rm = TRUE)
    recovery_2003 <- (post_drought_2003 - pre_drought_2003) / pre_drought_2003 * 100
    
    # Calculate overall resilience metrics
    mean_growth <- mean(ts_data$growth_rate, na.rm = TRUE)
    growth_variability <- sd(ts_data$growth_rate, na.rm = TRUE)
    cv <- growth_variability / mean_growth * 100
    
    return(data.frame(
        Species = species_name,
        Recovery_1976 = recovery_1976,
        Recovery_2003 = recovery_2003,
        Mean_Growth = mean_growth,
        Growth_Variability = growth_variability,
        Coefficient_Variation = cv
    ))
}

# Function for detailed drought analysis
detailed_drought_analysis <- function(ts_data, species_name) {
    # Calculate growth rates and moving averages
    ts_data$growth_rate <- c(NA, diff(ts_data$ring_width))
    ts_data$ma_5 <- forecast::ma(ts_data$ring_width, order = 5)
    ts_data$ma_10 <- forecast::ma(ts_data$ring_width, order = 10)
    
    # Identify drought years and their impact
    drought_years <- c(1976, 2003)
    drought_analysis <- data.frame()
    
    for (drought_year in drought_years) {
        # Define analysis window (5 years before and after)
        window_start <- drought_year - 5
        window_end <- drought_year + 5
        window_data <- ts_data[ts_data$year >= window_start & ts_data$year <= window_end, ]
        
        # Calculate pre-drought baseline (3 years before)
        pre_drought <- window_data[window_data$year < drought_year, ]
        baseline_growth <- mean(pre_drought$ring_width, na.rm = TRUE)
        baseline_growth_rate <- mean(pre_drought$growth_rate, na.rm = TRUE)
        
        # Calculate drought year impact
        drought_year_data <- window_data[window_data$year == drought_year, ]
        drought_impact <- (drought_year_data$ring_width - baseline_growth) / baseline_growth * 100
        
        # Calculate recovery period (3 years after)
        post_drought <- window_data[window_data$year > drought_year, ]
        recovery_growth <- mean(post_drought$ring_width[1:3], na.rm = TRUE)
        recovery_rate <- (recovery_growth - baseline_growth) / baseline_growth * 100
        
        # Calculate time to full recovery
        full_recovery_threshold <- baseline_growth * 0.95  # 95% of baseline
        recovery_years <- which(post_drought$ring_width >= full_recovery_threshold)
        time_to_recovery <- ifelse(length(recovery_years) > 0, recovery_years[1], NA)
        
        # Store results
        drought_result <- data.frame(
            Species = species_name,
            Drought_Year = drought_year,
            Baseline_Growth = baseline_growth,
            Drought_Impact_Pct = drought_impact,
            Recovery_Rate_Pct = recovery_rate,
            Time_to_Recovery_Years = time_to_recovery,
            Severity_Score = abs(drought_impact) / 100
        )
        
        drought_analysis <- rbind(drought_analysis, drought_result)
    }
    
    return(drought_analysis)
}

fit_arima_model <- function(ts_data, species_name) {
    # Check for stationarity
    adf_test <- adf.test(ts_data)
    print(paste("ADF Test p-value for", species_name, ":", adf_test$p.value))
    
    # Auto ARIMA model
    arima_model <- auto.arima(ts_data, seasonal = FALSE)
    print(paste("ARIMA model for", species_name, ":", arima_model))
    
    # Model diagnostics
    print(paste("AIC for", species_name, ":", AIC(arima_model)))
    print(paste("BIC for", species_name, ":", BIC(arima_model)))
    
    # Forecast next 10 years
    forecast_result <- forecast(arima_model, h = 10)
    
    return(list(
        model = arima_model,
        forecast = forecast_result,
        aic = AIC(arima_model),
        bic = BIC(arima_model)
    ))

# Function to create machine learning features
create_ml_features <- function(ts_data) {
    # Create lagged features
    ts_data$lag1 <- lag(ts_data$ring_width, 1)
    ts_data$lag2 <- lag(ts_data$ring_width, 2)
    ts_data$lag3 <- lag(ts_data$ring_width, 3)
    
    # Create moving averages
    ts_data$ma3 <- forecast::ma(ts_data$ring_width, order = 3)
    ts_data$ma5 <- forecast::ma(ts_data$ring_width, order = 5)
    
    # Create trend features
    ts_data$trend <- 1:nrow(ts_data)
    ts_data$trend_squared <- ts_data$trend^2
    
    # Create seasonal features
    ts_data$year_sin <- sin(2 * pi * ts_data$year / 10)
    ts_data$year_cos <- cos(2 * pi * ts_data$year / 10)
    
    # Create drought indicators
    ts_data$drought_1976 <- ifelse(ts_data$year == 1976, 1, 0)
    ts_data$drought_2003 <- ifelse(ts_data$year == 2003, 1, 0)
    
    # Create growth rate
    ts_data$growth_rate <- c(NA, diff(ts_data$ring_width))
    
    return(ts_data)
}

# Function to train Random Forest model
train_rf_model <- function(ml_data, species_name) {
    # Remove rows with NA values
    ml_data_clean <- na.omit(ml_data)
    
    # Prepare features and target
    features <- ml_data_clean[, c("lag1", "lag2", "lag3", "ma3", "ma5", 
                                 "trend", "trend_squared", "year_sin", "year_cos",
                                 "drought_1976", "drought_2003")]
    target <- ml_data_clean$ring_width
    
    # Split data for training and testing
    set.seed(123)
    train_index <- createDataPartition(target, p = 0.8, list = FALSE)
    train_features <- features[train_index, ]
    train_target <- target[train_index]
    test_features <- features[-train_index, ]
    test_target <- target[-train_index]
    
    # Train Random Forest model
    rf_model <- randomForest(train_features, train_target, ntree = 500, importance = TRUE)
    
    # Make predictions
    predictions <- predict(rf_model, test_features)
    
    # Calculate metrics
    rmse <- sqrt(mean((test_target - predictions)^2))
    mae <- mean(abs(test_target - predictions))
    r_squared <- cor(test_target, predictions)^2
    
    print(paste("Random Forest Model for", species_name))
    print(paste("RMSE:", round(rmse, 4)))
    print(paste("MAE:", round(mae, 4)))
    print(paste("R-squared:", round(r_squared, 4)))
    
    return(list(
        model = rf_model,
        rmse = rmse,
        mae = mae,
        r_squared = r_squared,
        predictions = predictions,
        actual = test_target
    ))
}
